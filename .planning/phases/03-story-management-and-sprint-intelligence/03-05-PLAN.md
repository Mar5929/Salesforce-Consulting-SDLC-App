**Burndown computation (src/lib/burndown.ts) -- NEW FILE:**
Per Pitfall 4 and research recommendation, compute burndown on-the-fly from Story.updatedAt:

```typescript
import { eachDayOfInterval, startOfDay, isBefore, isEqual } from "date-fns"

export interface BurndownPoint {
  date: string  // ISO date string for chart
  remaining: number | null  // null for future days
  ideal: number
}

export interface BurndownStory {
  storyPoints: number | null
  status: string
  updatedAt: Date
}

export function computeBurndown(
  sprintStart: Date,
  sprintEnd: Date,
  stories: BurndownStory[]
): BurndownPoint[] {
  const totalPoints = stories.reduce((sum, s) => sum + (s.storyPoints ?? 0), 0)
  const days = eachDayOfInterval({ start: startOfDay(sprintStart), end: startOfDay(sprintEnd) })
  const today = startOfDay(new Date())

  return days.map((day, i) => {
    const ideal = totalPoints - (totalPoints / Math.max(days.length - 1, 1)) * i

    // Future days: no actual data
    if (isBefore(today, day) && !isEqual(today, day)) {
      return { date: day.toISOString().slice(0, 10), remaining: null, ideal: Math.round(ideal * 10) / 10 }
    }

    // Count points completed by end of this day
    const completedByDay = stories
      .filter(s => s.status === "DONE" && (isBefore(startOfDay(s.updatedAt), day) || isEqual(startOfDay(s.updatedAt), day)))
      .reduce((sum, s) => sum + (s.storyPoints ?? 0), 0)

    return {
      date: day.toISOString().slice(0, 10),
      remaining: totalPoints - completedByDay,
      ideal: Math.round(ideal * 10) / 10,
    }
  })
}
```
