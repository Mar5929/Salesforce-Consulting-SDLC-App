```typescript
const getSessionTokenTotalsSchema = z.object({
  projectId: z.string().min(1),
  conversationId: z.string().min(1),
})

/**
 * Get aggregate token usage and cost for a conversation session.
 * Sums inputTokens + outputTokens and cost across all ChatMessage records.
 */
export const getSessionTokenTotals = actionClient
  .schema(getSessionTokenTotalsSchema)
  .action(async ({ parsedInput: { projectId, conversationId } }) => {
    await getCurrentMember(projectId)

    const result = await prisma.chatMessage.aggregate({
      where: { conversationId },
      _sum: {
        inputTokens: true,
        outputTokens: true,
        cost: true,
      },
    })

    return {
      totalTokens: (result._sum.inputTokens ?? 0) + (result._sum.outputTokens ?? 0),
      totalCost: result._sum.cost ?? 0,
    }
  })
```

Note: Ensure this uses the same `actionClient` and `getCurrentMember` patterns already in the file.

**1. Add accept/reject server actions to `src/actions/transcripts.ts`:**

Items are pre-created as DB records by agent harness tools during extraction. Accepting is a no-op status confirmation (items are already created). Rejecting should soft-delete by marking the entity. Since extracted entities are Questions, Decisions, Requirements (stored as Questions with scope), and Risks — and these don't have a universal "rejected" status — the simplest approach is:

For **reject**: Delete the record. The item was auto-created by AI and the user is explicitly rejecting it before it enters the workflow.

For **accept**: No-op (or set a confirmedAt timestamp if available). Since items are already persisted, accepting simply means keeping them.

Add these server actions:

```typescript
const extractionItemActionSchema = z.object({
  projectId: z.string().min(1),
  entityType: z.string().min(1),
  entityId: z.string().min(1),
})

/**
 * Accept an AI-extracted item. Items are already persisted by agent tools,
 * so accept is a no-op confirmation. Included for completeness and future
 * confirmed/unconfirmed tracking.
 */
export const acceptExtractionItem = actionClient
  .schema(extractionItemActionSchema)
  .action(async ({ parsedInput: { projectId, entityType, entityId } }) => {
    await getCurrentMember(projectId)
    // Items are already created in DB by agent tools. Accept = keep as-is.
    // Future: could set a `confirmedByUser` flag if schema supports it.
    return { accepted: true, entityType, entityId }
  })

/**
 * Reject an AI-extracted item by deleting it from the database.
 * The item was auto-created by AI during transcript processing;
 * rejecting removes it before it enters the project workflow.
 */
export const rejectExtractionItem = actionClient
  .schema(extractionItemActionSchema)
  .action(async ({ parsedInput: { projectId, entityType, entityId } }) => {
    await getCurrentMember(projectId)

    switch (entityType.toLowerCase()) {
      case "question":
        await prisma.question.delete({ where: { id: entityId } })
        break
      case "decision":
        await prisma.decision.delete({ where: { id: entityId } })
        break
      case "risk":
        await prisma.risk.delete({ where: { id: entityId } })
        break
      default:
        throw new Error(`Unknown extraction entity type: ${entityType}`)
    }

    return { rejected: true, entityType, entityId }
  })
```
