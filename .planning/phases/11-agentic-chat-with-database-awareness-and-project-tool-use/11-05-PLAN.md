    **Step 1: Complete src/lib/chat-tools/index.ts** — Uncomment all imports and wire the tool composition:

    ```typescript
    import type { ProjectRole } from "@/generated/prisma"
    import { ROLE_PERMISSIONS } from "./types"
    import { allQueryTools } from "./read"
    import { storyWriteTools, discoveryWriteTools, sprintWriteTools, defectWriteTools, testCaseWriteTools } from "./write"
    import { deleteAllTools } from "./delete"
    import { batchStoryTools, batchQuestionTools } from "./batch"

    export function buildToolsForRole(
      role: ProjectRole,
      projectId: string,
      memberId: string
    ) {
      const allowed = ROLE_PERMISSIONS[role] ?? ["query"]
      const tools: Record<string, unknown> = {}

      // Query tools — everyone gets these (D-10)
      Object.assign(tools, allQueryTools(projectId))

      // Write tools — gated by category (D-08)
      if (allowed.includes("mutate_story")) {
        Object.assign(tools, storyWriteTools(projectId, memberId, role))
        Object.assign(tools, batchStoryTools(projectId, memberId, role))
      }
      if (allowed.includes("mutate_discovery")) {
        Object.assign(tools, discoveryWriteTools(projectId, memberId))
        Object.assign(tools, batchQuestionTools(projectId, memberId))
      }
      if (allowed.includes("mutate_sprint")) {
        Object.assign(tools, sprintWriteTools(projectId, memberId))
      }
      if (allowed.includes("mutate_defect")) {
        Object.assign(tools, defectWriteTools(projectId, memberId))
      }
      if (allowed.includes("mutate_test_case")) {
        Object.assign(tools, testCaseWriteTools(projectId, memberId))
      }

      // Delete tools — only SA and PM (D-05, D-08)
      if (allowed.includes("delete")) {
        Object.assign(tools, deleteAllTools(projectId, memberId, role))
      }

      return tools
    }
    ```

    // For story sessions, merge the specialized story draft tool with agentic tools

      ? { ...agenticTools, create_story_draft: storyDraftTool }
