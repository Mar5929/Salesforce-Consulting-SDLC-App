    ```typescript
    import type { ProjectRole } from "@/generated/prisma"

    // Tool category for gating
    export type ToolCategory = "query" | "mutate_story" | "mutate_discovery" | "mutate_sprint" | "mutate_defect" | "mutate_test_case" | "delete"

    // Standardized tool result envelope
    export interface ToolResult<T = unknown> {
      success: boolean
      data?: T
      error?: string
      count?: number
    }

    // Summary-first pattern: compact entity representation for list queries (D-07)
    export interface EntitySummary {
      id: string
      displayId?: string
      title: string
      status?: string
      priority?: string
    }

    // Role-to-category permission map (D-08, D-09)
    // Based on existing server action role checks in stories.ts, questions.ts, etc.
    export const ROLE_PERMISSIONS: Record<ProjectRole, ToolCategory[]> = {
      SOLUTION_ARCHITECT: ["query", "mutate_story", "mutate_discovery", "mutate_sprint", "mutate_defect", "mutate_test_case", "delete"],
      PM: ["query", "mutate_story", "mutate_discovery", "mutate_sprint", "mutate_defect", "mutate_test_case", "delete"],
      BA: ["query", "mutate_story", "mutate_discovery", "mutate_defect", "mutate_test_case"],
      DEVELOPER: ["query", "mutate_story", "mutate_defect", "mutate_test_case"],
      QA: ["query", "mutate_defect", "mutate_test_case"],
    }

    // Helper: check if role has category
    export function roleHasCategory(role: ProjectRole, category: ToolCategory): boolean {
      return ROLE_PERMISSIONS[role]?.includes(category) ?? false
    }
    ```

    **src/lib/chat-tools/audit.ts** — Create audit logging helper (D-19):

    ```typescript
    import { prisma } from "@/lib/db"

    interface ToolCallRecord {
      toolName: string
      args: Record<string, unknown>
      result: unknown
      timestamp: string
    }

    // Log tool calls by updating the ChatMessage.toolCalls JSON field
    // Called from onFinish in the chat route after each AI turn
    export async function logToolCalls(params: {
      conversationId: string
      messageId: string
      toolCalls: ToolCallRecord[]
      totalInputTokens: number
      totalOutputTokens: number
      cost: number
    }) {
      if (params.toolCalls.length === 0) return

      await prisma.chatMessage.update({
        where: { id: params.messageId },
        data: {
          toolCalls: JSON.parse(JSON.stringify(params.toolCalls)),
          inputTokens: params.totalInputTokens,
          outputTokens: params.totalOutputTokens,
          cost: params.cost,
        },
      })
    }
    ```

    **src/lib/chat-tools/system-prompt.ts** — Build lean agentic system prompt (D-15):

    ```typescript
    import { prisma } from "@/lib/db"

    // Lean system prompt: ~500-800 tokens (D-15)
    // Tool descriptions teach the AI the data model (D-01)
    // AI discovers data via tools, not prompt dump
    export async function buildAgenticSystemPrompt(projectId: string): Promise<string> {
      const project = await prisma.project.findUnique({
        where: { id: projectId },
        select: {
          name: true,
          clientName: true,
          engagementType: true,
          currentPhase: true,
          description: true,
        },
      })

      if (!project) return "You are an AI project assistant. The project could not be loaded."

      return `You are an AI project assistant for a Salesforce consulting engagement. You have full access to query, create, update, and delete project entities through your tools.

## Project
- Name: ${project.name}
- Client: ${project.clientName}
- Engagement: ${project.engagementType}
- Phase: ${project.currentPhase ?? "Not set"}
${project.description ? `- Description: ${project.description}` : ""}

## Capabilities
You can search and retrieve any project data (epics, features, stories, questions, decisions, requirements, risks, sprints, defects, test cases, knowledge articles, org components, business processes, documents, conversations). You can create, update, and delete mutable entities. Use tools for data operations — do not fabricate entity IDs or data.

## Rules
- Always use query tools to look up data before answering questions about the project.
- For lists, use query tools (summary fields). For full details, use get_ tools.
- When creating or updating entities, confirm the action with the user in your response text.
- Delete operations require explicit user confirmation via the approval UI — never skip this.
- If a tool returns an error, explain what happened and suggest alternatives.
- Stay within this project's scope. Do not reference data from other projects.
- Be concise. Use Salesforce terminology correctly. Format with markdown.
- When performing multi-step workflows (e.g., create epic then generate stories), summarize progress after each step.`
    }
    ```
