    **src/lib/chat-tools/delete/delete-tools.ts** — All delete tools in one file (they're thin wrappers):

    ```typescript
    import { tool } from "ai"
    import { z } from "zod"
    import { prisma } from "@/lib/db"
    import { scopedPrisma } from "@/lib/project-scope"
    import type { ToolResult } from "../types"
    import type { ProjectRole } from "@/generated/prisma"

    export function deleteAllTools(projectId: string, memberId: string, role: ProjectRole) {
      const scoped = scopedPrisma(projectId)

      // Helper to create a delete tool with consistent pattern
      function makeDeleteTool(
        entityName: string,
        entityNameLower: string,
        modelName: keyof typeof prisma & string,
      ) {
        return tool({
          description: `Delete a ${entityName}. Requires explicit user confirmation before execution. This action is permanent.`,
          inputSchema: z.object({
            id: z.string().describe(`The ${entityNameLower} ID to delete`),
            reason: z.string().describe(`Why this ${entityNameLower} should be deleted`),
          }),
          needsApproval: true, // D-05: Hard-coded structural gate
          execute: async ({ id }): Promise<ToolResult> => {
            try {
              // Verify entity exists within project scope
              const entity = await (scoped[modelName] as any).findFirst({
                where: { id },
                select: { id: true },
              })
              if (!entity) {
                return { success: false, error: `${entityName} not found` }
              }

              // Execute deletion
              await (prisma[modelName] as any).delete({ where: { id } })
              return { success: true, data: { deletedId: id, entityType: entityName } }
            } catch (error) {
              return { success: false, error: error instanceof Error ? error.message : `Failed to delete ${entityNameLower}` }
            }
          },
        })
      }

      return {
        delete_epic: makeDeleteTool("Epic", "epic", "epic"),
        delete_feature: makeDeleteTool("Feature", "feature", "feature"),
        delete_story: makeDeleteTool("Story", "story", "story"),
        delete_question: makeDeleteTool("Question", "question", "question"),
        delete_decision: makeDeleteTool("Decision", "decision", "decision"),
        delete_requirement: makeDeleteTool("Requirement", "requirement", "requirement"),
        delete_risk: makeDeleteTool("Risk", "risk", "risk"),
        delete_sprint: makeDeleteTool("Sprint", "sprint", "sprint"),
        delete_defect: makeDeleteTool("Defect", "defect", "defect"),
        delete_test_case: makeDeleteTool("Test Case", "test case", "testCase"),
      }
    }
    ```
