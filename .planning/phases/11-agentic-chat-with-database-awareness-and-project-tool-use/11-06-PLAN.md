    **tool-display-names.ts** — Map tool names to user-facing labels:

    ```typescript
    const ENTITY_NAMES: Record<string, { singular: string; plural: string }> = {
      stories: { singular: "story", plural: "stories" },
      epics: { singular: "epic", plural: "epics" },
      features: { singular: "feature", plural: "features" },
      questions: { singular: "question", plural: "questions" },
      decisions: { singular: "decision", plural: "decisions" },
      requirements: { singular: "requirement", plural: "requirements" },
      risks: { singular: "risk", plural: "risks" },
      sprints: { singular: "sprint", plural: "sprints" },
      defects: { singular: "defect", plural: "defects" },
      test_cases: { singular: "test case", plural: "test cases" },
      articles: { singular: "article", plural: "articles" },
      org_components: { singular: "org component", plural: "org components" },
      business_processes: { singular: "business process", plural: "business processes" },
      documents: { singular: "document", plural: "documents" },
      conversations: { singular: "conversation", plural: "conversations" },
    }

    export function getToolDisplayName(toolName: string): string {
      // Extract entity from tool name pattern: action_entityPlural or action_entity
      const parts = toolName.split("_")
      const action = parts[0] // query, get, create, update, delete
      const entityKey = parts.slice(1).join("_") // stories, test_cases, etc.
      const entity = ENTITY_NAMES[entityKey]

      if (!entity) return toolName.replace(/_/g, " ")

      switch (action) {
        case "query": return `Searching ${entity.plural}`
        case "get": return `Loading ${entity.singular} details`
        case "create": return entityKey === "stories" || entityKey === "questions" ? `Creating multiple ${entity.plural}` : `Creating ${entity.singular}`
        case "update": return `Updating ${entity.singular}`
        case "delete": return `Preparing to delete ${entity.singular}`
        default: return toolName.replace(/_/g, " ")
      }
    }

    export function getToolCategory(toolName: string): "query" | "detail" | "create" | "update" | "delete" | "batch" | "unknown" {
      if (toolName.startsWith("query_")) return "query"
      if (toolName.startsWith("get_")) return "detail"
      if (toolName.startsWith("delete_")) return "delete"
      if (toolName.startsWith("update_")) return "update"
      if (toolName.startsWith("create_")) {
        // Batch tools: create_stories, create_questions
        if (toolName === "create_stories" || toolName === "create_questions") return "batch"
        return "create"
      }
      return "unknown"
    }
    ```

    **src/components/chat/tool-part-renderer.tsx** — Central dispatcher for tool invocation parts:

    Read the actual UIMessage part types from @ai-sdk/react to understand the current part structure. The AI SDK v6 uses typed parts on messages. Tool invocations appear as parts with tool-related types.

    The renderer should handle these states for each tool invocation:
    1. **Loading** (state is partial-call or call without result): Render ToolLoadingIndicator
    2. **Success result**: Route to appropriate card based on tool category:
       - query_* -> EntityQueryCard
       - get_* -> EntityDetailCard
       - create_* / update_* -> MutationConfirmCard
       - delete_* -> not reached here (delete tools use approval flow)
       - create_story_draft -> existing StoryDraftCards (preserve backward compat)
    3. **Error result** (result.success === false): Render ToolErrorCard
    4. **Approval requested** (state indicates pending approval for delete tools): Render DeleteConfirmationCard

    ```typescript
    "use client"

    import { getToolCategory } from "./tool-cards/tool-display-names"
    import { EntityQueryCard } from "./tool-cards/entity-query-card"
    import { EntityDetailCard } from "./tool-cards/entity-detail-card"
    import { MutationConfirmCard } from "./tool-cards/mutation-confirm-card"
    import { DeleteConfirmationCard } from "./tool-cards/delete-confirmation-card"
    import { ToolLoadingIndicator } from "./tool-cards/tool-loading-indicator"
    import { ToolErrorCard } from "./tool-cards/tool-error-card"

    interface ToolPartRendererProps {
      toolName: string
      toolCallId: string
      args: Record<string, unknown>
      state: string
      result?: unknown
      onApprove?: (toolCallId: string) => void
      onDeny?: (toolCallId: string) => void
    }

    export function ToolPartRenderer({ toolName, toolCallId, args, state, result, onApprove, onDeny }: ToolPartRendererProps) {
      // Skip create_story_draft — handled by StoryDraftCards in message-list
      if (toolName === "create_story_draft") return null

      // Loading states
      if (state === "partial-call" || (state === "call" && !result)) {
        return <ToolLoadingIndicator toolName={toolName} />
      }

      // Approval requested (delete tools with needsApproval)
      if (state === "approval-requested") {
        return (
          <DeleteConfirmationCard
            toolName={toolName}
            args={args as { id: string; reason: string }}
            approvalId={toolCallId}
            onApprove={(id) => onApprove?.(id)}
            onDeny={(id) => onDeny?.(id)}
          />
        )
      }

      // Result available
      if (result) {
        const typedResult = result as { success?: boolean; error?: string; data?: unknown; count?: number }

        // Error result
        if (typedResult.success === false && typedResult.error) {
          return <ToolErrorCard toolName={toolName} error={typedResult.error} />
        }

        // Route by tool category
        const category = getToolCategory(toolName)
        switch (category) {
          case "query":
            return <EntityQueryCard toolName={toolName} result={typedResult} />
          case "detail":
            return <EntityDetailCard toolName={toolName} result={typedResult} />
          case "create":
          case "update":
          case "batch":
            return <MutationConfirmCard toolName={toolName} result={typedResult} />
          default:
            return <MutationConfirmCard toolName={toolName} result={typedResult} />
        }
      }

      return <ToolLoadingIndicator toolName={toolName} />
    }
    ```
