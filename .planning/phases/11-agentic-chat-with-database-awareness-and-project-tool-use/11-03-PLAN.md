    **src/lib/chat-tools/write/mutate-stories.ts:**
    ```typescript
    import { tool } from "ai"
    import { z } from "zod"
    import { scopedPrisma } from "@/lib/project-scope"
    import { sanitizeToolInput } from "@/lib/agent-harness/sanitize"
    import { createId } from "@paralleldrive/cuid2"

    // generateDisplayId helper: read how existing stories.ts generates displayId.
    // If it uses a counter-based approach, replicate it. If it uses createId, use that.
    // Fallback: prefix + createId().slice(0, 8).toUpperCase()

    export function mutateStoriesTools(projectId: string, memberId: string, role: string) {
      const scoped = scopedPrisma(projectId)

      return {
        create_story: tool({
          description: "Create a new user story. Returns the created story summary.",
          inputSchema: z.object({
            epicId: z.string().describe("Epic this story belongs to"),
            featureId: z.string().optional().describe("Feature this story belongs to (optional)"),
            title: z.string().min(1).max(200),
            persona: z.string().optional(),
            description: z.string().optional(),
            acceptanceCriteria: z.string().optional(),
            storyPoints: z.number().int().min(1).max(13).optional(),
            priority: z.enum(["LOW","MEDIUM","HIGH","CRITICAL"]).default("MEDIUM"),
            assigneeId: z.string().optional().describe("ProjectMember ID to assign"),
          }),
          execute: async (input) => {
            try {
              const s = sanitizeToolInput(input)
              // Verify epic belongs to project
              const epic = await scoped.epic.findUnique({ where: { id: s.epicId }, select: { id: true, prefix: true } })
              if (!epic) return { success: false, error: "Epic not found in this project" }
              // Generate displayId — read existing stories to determine current max and increment,
              // or use a simple prefix pattern if no counter exists
              const count = await scoped.story.count()
              const displayId = `${epic.prefix}-STR-${String(count + 1).padStart(3, "0")}`
              const story = await scoped.story.create({
                data: {
                  epicId: s.epicId,
                  featureId: s.featureId,
                  displayId,
                  title: s.title,
                  persona: s.persona,
                  description: s.description,
                  acceptanceCriteria: s.acceptanceCriteria,
                  storyPoints: s.storyPoints,
                  priority: s.priority,
                  status: "DRAFT",
                  assigneeId: s.assigneeId,
                },
                select: { id: true, displayId: true, title: true, status: true, priority: true },
              })
              return { success: true, story }
            } catch (e) {
              return { success: false, error: e instanceof Error ? e.message : "Create failed" }
            }
          },
        }),

        update_story: tool({
          description: "Update fields on an existing story. Only provide fields to change.",
          inputSchema: z.object({
            storyId: z.string(),
            title: z.string().min(1).max(200).optional(),
            description: z.string().optional(),
            acceptanceCriteria: z.string().optional(),
            storyPoints: z.number().int().min(1).max(13).optional(),
            priority: z.enum(["LOW","MEDIUM","HIGH","CRITICAL"]).optional(),
            status: z.enum(["DRAFT","READY","SPRINT_PLANNED","IN_PROGRESS","IN_REVIEW","QA","DONE"]).optional(),
            assigneeId: z.string().optional(),
          }),
          execute: async (input) => {
            try {
              const s = sanitizeToolInput(input)
              const existing = await scoped.story.findUnique({ where: { id: s.storyId }, select: { id: true, assigneeId: true } })
              if (!existing) return { success: false, error: "Story not found" }
              // Role check: BA and DEVELOPER can only update own stories
              if (["BA","DEVELOPER"].includes(role) && existing.assigneeId !== memberId) {
                return { success: false, error: "You can only update stories assigned to you" }
              }
              const { storyId, ...updateData } = s
              const story = await scoped.story.update({
                where: { id: storyId },
                data: updateData,
                select: { id: true, displayId: true, title: true, status: true },
              })
              return { success: true, story }
            } catch (e) {
              return { success: false, error: e instanceof Error ? e.message : "Update failed" }
            }
          },
        }),
      }
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| AI tool input → DB write | sanitizeToolInput(input) is the mandatory first call in every execute handler |
| role string → tool availability | role comes from getCurrentMember() server-side; never from request body |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-08 | Tampering | All write tool execute handlers | mitigate | `sanitizeToolInput(input)` called as first line — strips HTML/script from all string fields via DOMPurify before any DB write |
| T-11-09 | Elevation of Privilege | mutate-stories update_story | mitigate | Role check inside execute: BA/DEVELOPER blocked from updating stories not assigned to them (assigneeId !== memberId guard) |
| T-11-10 | Elevation of Privilege | mutate-sprints | mitigate | Included in tools object only when `role === "PM"` — structural gate, not prompt-based |
| T-11-11 | Denial of Service | batch tools | mitigate | `z.array(...).max(20)` hard cap on batch size — prevents runaway bulk creates |
| T-11-12 | Tampering | Risk severity computation | accept | Severity is computed server-side from likelihood x impact matrix — AI cannot inject a severity value directly |
</threat_model>
