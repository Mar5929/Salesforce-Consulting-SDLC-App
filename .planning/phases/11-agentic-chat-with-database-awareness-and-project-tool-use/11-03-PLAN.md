    ```typescript
    // src/lib/chat-tools/write/mutate-stories.ts
    import { tool } from "ai"
    import { z } from "zod"
    import { scopedPrisma } from "@/lib/project-scope"
    import { sanitizeToolInput } from "@/lib/agent-harness/sanitize"
    import { generateStoryDisplayId } from "@/lib/display-id"
    import { canTransition } from "@/lib/story-status-machine"
    import { createId } from "@paralleldrive/cuid2"
    import type { ToolResult } from "../types"
    import type { ProjectRole } from "@/generated/prisma"

    export function mutateStoryTools(projectId: string, memberId: string, role: ProjectRole) {
      return {
        create_story: tool({
          description: "Create a new user story in an epic. Returns the created story with ID and displayId.",
          inputSchema: z.object({
            epicId: z.string().describe("Epic ID this story belongs to"),
            featureId: z.string().optional().describe("Feature ID within the epic"),
            title: z.string().min(1).max(200).describe("Story title"),
            persona: z.string().optional().describe("User persona, e.g. 'Sales Manager'"),
            description: z.string().optional().describe("Story description in user story format"),
            acceptanceCriteria: z.string().min(1).describe("Acceptance criteria (Given/When/Then)"),
            storyPoints: z.number().int().min(0).max(100).optional().describe("Fibonacci estimate"),
            priority: z.enum(["LOW", "MEDIUM", "HIGH", "CRITICAL"]).default("MEDIUM"),
          }),
          execute: async (input): Promise<ToolResult> => {
            try {
              const sanitized = sanitizeToolInput(input)
              const scoped = scopedPrisma(projectId)
              const id = createId()
              const displayId = await generateStoryDisplayId(projectId)

              const story = await scoped.story.create({
                data: {
                  id,
                  displayId,
                  projectId,
                  epicId: sanitized.epicId,
                  featureId: sanitized.featureId ?? null,
                  title: sanitized.title,
                  persona: sanitized.persona ?? null,
                  description: sanitized.description ?? null,
                  acceptanceCriteria: sanitized.acceptanceCriteria,
                  storyPoints: sanitized.storyPoints ?? null,
                  priority: sanitized.priority ?? "MEDIUM",
                  status: "DRAFT",
                  createdById: memberId,
                },
                select: { id: true, displayId: true, title: true, status: true, priority: true },
              })
              return { success: true, data: story }
            } catch (error) {
              return { success: false, error: error instanceof Error ? error.message : "Failed to create story" }
            }
          },
        }),

        update_story: tool({
          description: "Update an existing user story. Only provided fields are changed.",
          inputSchema: z.object({
            storyId: z.string().describe("Story ID to update"),
            title: z.string().min(1).max(200).optional().describe("New title"),
            persona: z.string().optional().describe("New persona"),
            description: z.string().optional().describe("New description"),
            acceptanceCriteria: z.string().optional().describe("New acceptance criteria"),
            storyPoints: z.number().int().min(0).max(100).optional().describe("New story points"),
            priority: z.enum(["LOW", "MEDIUM", "HIGH", "CRITICAL"]).optional(),
            status: z.enum(["DRAFT", "READY", "SPRINT_PLANNED", "IN_PROGRESS", "IN_REVIEW", "QA", "DONE"]).optional().describe("New status (must be a valid transition)"),
            assigneeId: z.string().nullable().optional().describe("New assignee member ID"),
          }),
          execute: async (input): Promise<ToolResult> => {
            try {
              const sanitized = sanitizeToolInput(input)
              const scoped = scopedPrisma(projectId)

              // Verify story exists and check permissions
              const existing = await scoped.story.findFirst({ where: { id: sanitized.storyId } })
              if (!existing) return { success: false, error: "Story not found" }

              // Role check: non-PM/SA can only edit own stories
              if (!["PM", "SOLUTION_ARCHITECT"].includes(role) && existing.createdById !== memberId && existing.assigneeId !== memberId) {
                return { success: false, error: "You can only edit stories you created or are assigned to" }
              }

              // Status transition validation
              if (sanitized.status && sanitized.status !== existing.status) {
                if (!canTransition(existing.status, sanitized.status as never)) {
                  return { success: false, error: `Cannot transition from ${existing.status} to ${sanitized.status}` }
                }
              }

              const data: Record<string, unknown> = {}
              if (sanitized.title !== undefined) data.title = sanitized.title
              if (sanitized.persona !== undefined) data.persona = sanitized.persona
              if (sanitized.description !== undefined) data.description = sanitized.description
              if (sanitized.acceptanceCriteria !== undefined) data.acceptanceCriteria = sanitized.acceptanceCriteria
              if (sanitized.storyPoints !== undefined) data.storyPoints = sanitized.storyPoints
              if (sanitized.priority !== undefined) data.priority = sanitized.priority
              if (sanitized.status !== undefined) data.status = sanitized.status
              if (sanitized.assigneeId !== undefined) data.assigneeId = sanitized.assigneeId

              const updated = await scoped.story.update({
                where: { id: sanitized.storyId },
                data,
                select: { id: true, displayId: true, title: true, status: true, priority: true },
              })
              return { success: true, data: updated }
            } catch (error) {
              return { success: false, error: error instanceof Error ? error.message : "Failed to update story" }
            }
          },
        }),
      }
    }
    ```

    ```typescript
    import { mutateQuestionTools } from "./mutate-questions"
    ```
