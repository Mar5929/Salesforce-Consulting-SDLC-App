```typescript
import { tool } from "ai"
import { z } from "zod"

// needsApproval: true is the ONLY structural deletion gate (D-05).
// The execute function only runs AFTER the user explicitly approves via the UI.
// The AI cannot bypass this by any prompt technique.
export function deleteTools(projectId: string, memberId: string, role: string) {
  const scoped = scopedPrisma(projectId)
  return {
    delete_story: tool({
      description: "Delete a user story permanently. Requires explicit user confirmation.",
      inputSchema: z.object({
        storyId: z.string().describe("ID of the story to delete"),
        reason: z.string().describe("Why this story should be deleted"),
      }),
      needsApproval: true,  // <-- D-05 structural gate
      execute: async ({ storyId, reason: _reason }) => {
        try {
          // Double-check role at execute time (approval could be replayed)
          if (!["SOLUTION_ARCHITECT", "PM"].includes(role)) {
            return { success: false, error: "Insufficient permissions to delete stories" }
          }
          const story = await scoped.story.findUnique({ where: { id: storyId }, select: { id: true, title: true, displayId: true } })
          if (!story) return { success: false, error: "Story not found" }
          await scoped.story.delete({ where: { id: storyId } })
          return { success: true, deleted: { id: storyId, displayId: story.displayId, title: story.title } }
        } catch (e) {
          return { success: false, error: e instanceof Error ? e.message : "Delete failed" }
        }
      },
    }),
    // ... repeat for other entities
  }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| needsApproval gate | AI SDK structural gate — execute only runs after addToolApprovalResponse(approved: true) from client |
| execute role double-check | Role verified at execute time, not just at tool registration time, to prevent approval replay |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-13 | Elevation of Privilege | All delete tools | mitigate | `needsApproval: true` is a structural code-level gate (D-05) — the AI cannot cause deletion via any prompt technique |
| T-11-14 | Elevation of Privilege | delete execute handlers | mitigate | Role double-check inside execute: `if (!["SOLUTION_ARCHITECT","PM"].includes(role)) return { success: false, error: "..." }` — prevents approval replay by a lower-privileged role |
| T-11-15 | Information Disclosure | delete execute response | accept | Delete result only returns `{ id, displayId, title }` — no sensitive field data leaks in the tool result |
| T-11-16 | Tampering | delete tool inputs | mitigate | `sanitizeToolInput(input)` called first in execute — reason field stripped of HTML/script |
| T-11-17 | Denial of Service | Cascading deletes (Epic delete) | accept | Epic deletion cascades via Prisma `onDelete: Cascade` — user is warned in tool description ("This will also delete all features and stories under this epic.") |
</threat_model>
