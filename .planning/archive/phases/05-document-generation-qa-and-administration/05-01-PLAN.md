   import { prisma } from "./db"

   export async function assertProjectNotArchived(projectId: string): Promise<void> {
     const project = await prisma.project.findUniqueOrThrow({
       where: { id: projectId },
       select: { status: true },
     })
     if (project.status === "ARCHIVED") {
       throw new Error("This project is archived. Reactivate to make changes.")

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | T (Tampering) | s3-storage.ts | mitigate | S3 keys constructed server-side from projectId + cuid, never user-supplied. ContentType derived from format enum, not user input. |
| T-05-02 | I (Info Disclosure) | s3-storage.ts | mitigate | Presigned URLs expire in 5 minutes. s3Key never exposed to client -- only presigned URL returned. |
| T-05-03 | E (Elevation) | archive-guard.ts | mitigate | assertProjectNotArchived checks project.status at DB level before every mutation. |
