**Agent harness task (src/lib/agent-harness/tasks/sprint-intelligence.ts) -- NEW FILE:**
Task definition for sprint conflict analysis:
- id: `"sprint-intelligence"`
- systemPrompt: Instructs the AI to:
  1. Analyze component overlaps between stories
  2. For each overlap, assess severity:
     - HIGH: Two stories both MODIFY the same component, or one CREATEs and another DELETEs
     - MEDIUM: Two stories both CREATE on the same component (potential naming conflicts)
     - LOW: One MODIFIES and one CREATEs the same component (likely compatible)
  3. Suggest a dependency ordering for stories based on component relationships
  4. Explain reasoning concisely
  5. Output a JSON object matching the SprintAnalysis interface:
     ```json
     {
       "conflicts": [{ "storyADisplayId": "...", "storyBDisplayId": "...", "componentName": "...", "impactA": "...", "impactB": "...", "severity": "HIGH", "reasoning": "..." }],
       "dependencies": [{ "order": 1, "storyDisplayId": "...", "reasoning": "..." }]
     }
     ```
- contextSources: `[storiesInSprintContext]`
- tools: none (output-only analysis)
- maxTokens: 2000
- temperature: 0.3 (deterministic analysis)

```typescript
import { inngest } from "../client"
import { EVENTS } from "../events"
import { prisma } from "@/lib/db"
import Anthropic from "@anthropic-ai/sdk"

export const sprintIntelligence = inngest.createFunction(
  { id: "sprint-intelligence", retries: 2 },
  { event: EVENTS.SPRINT_STORIES_CHANGED },
  async ({ event, step }) => {
    const { sprintId, projectId } = event.data

    // Step 1: Load sprint stories with components
    const sprintData = await step.run("load-sprint-data", async () => {
      const stories = await prisma.story.findMany({
        where: { sprintId, projectId },
        select: {
          id: true,
          displayId: true,
          title: true,
          storyComponents: {
            select: { componentName: true, impactType: true },
          },
        },
      })
      return stories
    })

    // Step 2: Find component overlaps (deterministic, no AI)
    const overlaps = await step.run("detect-overlaps", async () => {
      const componentMap: Record<string, Array<{ displayId: string; title: string; impactType: string }>> = {}
      for (const story of sprintData) {
        for (const comp of story.storyComponents) {
          if (!comp.componentName) continue
          if (!componentMap[comp.componentName]) componentMap[comp.componentName] = []
          componentMap[comp.componentName].push({
            displayId: story.displayId,
            title: story.title,
            impactType: comp.impactType,
          })
        }
      }
      // Return only components appearing in 2+ stories
      return Object.entries(componentMap)
        .filter(([, stories]) => stories.length > 1)
        .map(([name, stories]) => ({ componentName: name, stories }))
    })

    if (overlaps.length === 0) {
      // No conflicts -- store empty result
      await step.run("store-empty-results", async () => {
        await prisma.sprint.update({
          where: { id: sprintId },
          data: {
            cachedAnalysis: {
              conflicts: [],
              dependencies: [],
              analyzedAt: new Date().toISOString(),
            },
          },
        })
      })
      return { conflicts: 0, dependencies: 0 }
    }

    // Step 3: AI analysis of overlaps
    const analysis = await step.run("ai-analysis", async () => {
      const anthropic = new Anthropic()
      const contextText = overlaps.map(o =>
        `Component "${o.componentName}": ${o.stories.map(s => `${s.displayId} "${s.title}" (${s.impactType})`).join(", ")}`
      ).join("\n")

      const allStories = sprintData.map(s =>
        `${s.displayId}: "${s.title}" [${s.storyComponents.map(c => `${c.componentName} (${c.impactType})`).join(", ")}]`
      ).join("\n")

      const response = await anthropic.messages.create({
        model: "claude-sonnet-4-20250514",
        max_tokens: 1500,
        temperature: 0.3,
        system: `You are a sprint planning advisor for a Salesforce project. Analyze component overlaps between user stories and provide conflict severity assessments and dependency ordering suggestions. Output valid JSON only.`,
        messages: [{
          role: "user",
          content: `## Component Overlaps\n${contextText}\n\n## All Stories\n${allStories}\n\nAnalyze these overlaps. For each, rate severity (HIGH/MEDIUM/LOW) based on impact types. Then suggest an execution ordering for ALL stories. Output JSON:\n{"conflicts": [{"storyADisplayId": "...", "storyBDisplayId": "...", "componentName": "...", "impactA": "...", "impactB": "...", "severity": "...", "reasoning": "..."}], "dependencies": [{"order": 1, "storyDisplayId": "...", "reasoning": "..."}]}`,
        }],
      })

      const text = response.content[0].type === "text" ? response.content[0].text : ""
      // Extract JSON from response (handle markdown code blocks)
      const jsonMatch = text.match(/\{[\s\S]*\}/)
      if (!jsonMatch) return { conflicts: [], dependencies: [] }
      return JSON.parse(jsonMatch[0]) as { conflicts: Array<Record<string, unknown>>; dependencies: Array<Record<string, unknown>> }
    })

    // Step 4: Store results
    await step.run("store-results", async () => {
      const cachedAnalysis = {
        conflicts: (analysis.conflicts || []).map((c: Record<string, unknown>, i: number) => ({
          id: `conflict-${i}`,
          storyA: { displayId: c.storyADisplayId, title: "" },
          storyB: { displayId: c.storyBDisplayId, title: "" },
          componentName: c.componentName,
          impactA: c.impactA,
          impactB: c.impactB,
          severity: c.severity || "MEDIUM",
          reasoning: c.reasoning || "",
          dismissed: false,
        })),
        dependencies: (analysis.dependencies || []).map((d: Record<string, unknown>) => ({
          order: d.order,
          storyDisplayId: d.storyDisplayId,
          storyTitle: "",
          reasoning: d.reasoning || "",
        })),
        analyzedAt: new Date().toISOString(),
      }
      await prisma.sprint.update({
        where: { id: sprintId },
        data: { cachedAnalysis },
      })

      // Send notification if conflicts detected
      if (cachedAnalysis.conflicts.length > 0) {
        await inngest.send({
          name: EVENTS.NOTIFICATION_SEND,
          data: {
            projectId,
            type: "SPRINT_CONFLICT_DETECTED",
            entityType: "SPRINT",
            entityId: sprintId,
            title: `${cachedAnalysis.conflicts.length} component conflict(s) detected in sprint`,
          },
        })
      }
    })
  }
)
```

```typescript
"use client"
import { useState } from "react"
import { AlertTriangle, ChevronDown, ChevronUp, X } from "lucide-react"
import { Badge } from "@/components/ui/badge"
import { Button } from "@/components/ui/button"

interface Conflict {
  id: string
  storyA: { displayId: string; title: string }
  storyB: { displayId: string; title: string }
  componentName: string
  impactA: string
  impactB: string
  severity: string
  reasoning: string
  dismissed: boolean
}

interface ConflictBannerProps {
  conflicts: Conflict[]
  onDismiss?: (conflictId: string) => void
}

export function ConflictBanner({ conflicts, onDismiss }: ConflictBannerProps) {
  const [expanded, setExpanded] = useState(false)
  const activeConflicts = conflicts.filter(c => !c.dismissed)

  if (activeConflicts.length === 0) return null

  const severityColor = {
    HIGH: "bg-red-50 border-red-200 text-red-800",
    MEDIUM: "bg-yellow-50 border-yellow-200 text-yellow-800",
    LOW: "bg-blue-50 border-blue-200 text-blue-800",
  }

  return (
    <div className="rounded-lg border border-yellow-200 bg-yellow-50 p-3 mb-4">
      <div className="flex items-center justify-between cursor-pointer" onClick={() => setExpanded(!expanded)}>
        <div className="flex items-center gap-2">
          <AlertTriangle className="h-4 w-4 text-yellow-600" />
          <span className="text-sm font-medium text-yellow-800">
            {activeConflicts.length} component conflict{activeConflicts.length > 1 ? "s" : ""} detected
          </span>
        </div>
        {expanded ? <ChevronUp className="h-4 w-4" /> : <ChevronDown className="h-4 w-4" />}
      </div>
      {expanded && (
        <div className="mt-3 space-y-2">
          {activeConflicts.map(conflict => (
            <div key={conflict.id} className={`flex items-start justify-between rounded-md border p-2 text-xs ${severityColor[conflict.severity as keyof typeof severityColor] || severityColor.MEDIUM}`}>
              <div>
                <div className="font-medium">
                  {conflict.storyA.displayId} & {conflict.storyB.displayId} overlap on "{conflict.componentName}"
                </div>
                <div className="mt-0.5 text-muted-foreground">{conflict.reasoning}</div>
                <Badge variant="outline" className="mt-1">{conflict.severity}</Badge>
              </div>
              {onDismiss && (
                <Button variant="ghost" size="sm" className="h-6 w-6 p-0" onClick={(e) => { e.stopPropagation(); onDismiss(conflict.id) }}>
                  <X className="h-3 w-3" />
                </Button>
              )}
            </div>
          ))}
        </div>
      )}
    </div>
  )
}
```
